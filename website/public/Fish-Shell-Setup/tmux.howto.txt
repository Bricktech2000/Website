// https://github.com/craftzdog/dotfiles-public/blob/master/.tmux.conf
// https://superuser.com/questions/253786/how-can-i-make-tmux-use-my-default-shell
// https://superuser.com/questions/209437/how-do-i-scroll-in-tmux
vim ~/.tmux.conf
paste:
set -g default-terminal "xterm-256color"
set-option -g default-shell /usr/bin/fish
setw -g xterm-keys off

# Moving window
bind-key -n M-S-y swap-window -t -1 \; previous-window
bind-key -n M-S-o swap-window -t +1 \; next-window
bind-key -n M-y previous-window \; send Escape
bind-key -n M-o next-window \; send Escape
bind-key -n M-i new-window \; send Escape
bind-key -n M-u kill-window \; send Escape

set-option -g status-justify "left"
set-window-option -g mode-keys vi
set -g mouse on # shift-click to select

set-option -g status-bg black
set-option -g history-limit 64096

set -sg escape-time 10

# default statusbar colors
set-option -g status-style bg=colour235,fg=colour136,default
# default window title colors
set-window-option -g window-status-style fg=colour244,bg=colour234,dim
# active window title colors
set-window-option -g window-status-current-style fg=colour166,bg=default,bright
# message text
set-option -g message-style bg=colour235,fg=colour166
# pane number display
set-option -g display-panes-active-colour colour33 #blue
set-option -g display-panes-colour colour166 #orange

# allow the title bar to adapt to whatever host you connect to
set -g set-titles on
set -g set-titles-string "#T"

# Powerline
source ~/.tmux.powerline.conf


// https://github.com/craftzdog/dotfiles-public/blob/master/.tmux.powerline.conf
vim ~/.tmux.powerline.conf
# vim: ft=tmux

if-shell 'test -z "$POWERLINE_CONFIG_COMMAND"' 'set-environment -g POWERLINE_CONFIG_COMMAND powerline-config'

# Don't version-check for this core functionality -- anything too old to
# support these options likely won't work well with powerline
set -g status on
set -g status-interval 2
set -g status-left-length 60
set -g status-left '#[fg=colour235,bg=colour252,bold] #S #[fg=colour252,bg=colour238,nobold]#[fg=colour245,bg=colour238,bold] #(whoami) #[fg=colour238,bg=colour234,nobold]'

set -g status-right-length 150
set -g status-right '#[fg=colour59]#[fg=colour255,bg=colour59] #[fg=colour254,bold]#[fg=colour16,bg=colour254,bold] #h '
